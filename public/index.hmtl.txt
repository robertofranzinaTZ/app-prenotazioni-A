  button { padding: 5px 10px; }

<body>
<h1>Prenotazioni</h1>
<table id="slots-table">
  <thead>
    <tr id="header-row"></tr>
  </thead>
  <tbody id="body-slots"></tbody>
</table>

<script>
async function loadSlots() {
  const res = await fetch("/api/slots");
  const slots = await res.json();

  // Crea intestazioni dinamiche
  const giorni = [...new Set(slots.map(s => s.giorno))];
  const headerRow = document.getElementById("header-row");
  headerRow.innerHTML = "<th>Ora</th>" + giorni.map(g => `<th>${g}</th>`).join("");

  // Raggruppa per ora
  const ore = [...new Set(slots.map(s => s.ora))];
  const tbody = document.getElementById("body-slots");
  tbody.innerHTML = "";

  for (let ora of ore) {
    const tr = document.createElement("tr");
    tr.innerHTML = `<td>${ora}</td>`;
    for (let giorno of giorni) {
      const slot = slots.find(s => s.ora === ora && s.giorno === giorno);
      const posti = slot ? slot.posti_totali - slot.posti_prenotati : 0;
      const btn = document.createElement("button");
      btn.textContent = posti > 0 ? `${posti} posti` : "Completo";
      btn.disabled = posti <= 0;
      btn.onclick = () => prenota(slot);
      const td = document.createElement("td");
      td.appendChild(btn);
      tr.appendChild(td);
    }
    tbody.appendChild(tr);
  }
}

async function prenota(slot) {
  const nome = prompt("Inserisci il tuo nome:");
  if (!nome) return;

  const res = await fetch("/api/prenota", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ ora: slot.ora, giorno: slot.giorno, nome })
  });
  const data = await res.json();
  if (data.success) {
    alert(`Prenotazione confermata! Posti restanti: ${data.posti_restanti}`);
    loadSlots();
  } else {
    alert(data.error || "Errore nella prenotazione");
  }
}

loadSlots();
</script>
</body>
</html>
